country_event = { #Oman Civil War End- Qaboos Victory
	id = OMA.6
	immediate = { log = "[GetDateText]: [Root.GetName]: event OMA.6" }
	title = OMA.6.t
	desc = OMA.6.d
	picture = GFX_report_event_generic_map

	fire_only_once = yes
	is_triggered_only = yes

	option = {
		name = OMA.6.a
		ITA = { set_autonomy = { target = OMA autonomy_state = autonomy_italy_hefty_influence } }
		#Make Oman into an italian associated state.
		OMA = { 
			set_politics = { ruling_party = despotism }
			set_popularities = {
				communist = 5
				socialist = 5
				progressivism = 0
				liberalism = 0
				conservatism = 0
				paternalism = 10
				despotism = 80
				fascism = 0
				ultranationalism = 0
			}
		}

		#Then- if italy is in a faction- join that faction.
		if = {
			limit = { ITA = { is_in_faction_with = USA } }
			USA = { add_to_faction = OMA }
			OMA = { add_ideas = OFN_Independent_Member }
		}
		else_if = {
			limit = { ITA = { is_in_faction_with = JAP } }
			JAP = { add_to_faction = OMA }
			OMA = { add_ideas = Sphere_Coequal }
		}
		else_if = {
			limit = { ITA = { is_in_faction = yes } }
			ITA = { add_to_faction = OMA }
			OMA = {
				if = {
					limit = { ITA = { has_idea = ITA_Mediterranean_Bloc_Leader } }
					add_ideas = ITA_Mediterranean_Bloc_Member
				}
				if = {
					limit = { ITA = { has_idea = ITA_Imperial_Alliance_Leader } }
					add_ideas = ITA_Imperial_Alliance_Member
				}
				if = {
					limit = { ITA = { has_idea = ITA_Rome_Pact_Leader } }
					add_ideas = ITA_Rome_Pact_Member
				}
			}
		}
		
		OMB = { remove_state_claim = 1336 }
		OMI = { remove_state_claim = 1336 }
		remove_state_claim = 1336
		add_state_core = 1336
		news_event = { id = WORLD.18028 days = 1 }
	}
}
