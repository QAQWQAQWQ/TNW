scripted_gui = {
	# # Main
	TNO_Army_Help_Button = {
		window_name = "Army_HelpButton"
		context_type = player_context
		parent_window_token = deployment_tab

		visible = {
			NOT = {
				has_global_flag = TNO_my_world_is_on_fire_how_about_yours
			}
		}

		effects = {
			army_help_toggle_click = {
				if = {
					limit = {
						has_country_flag = TNO_Tips_And_Tricks_Open
					}
					clr_country_flag = TNO_Tips_And_Tricks_Open
				}
				else = { 
					set_country_flag = TNO_Tips_And_Tricks_Open 
					set_variable = { current_tip_id = 12 } # sets the page to Economy section intro
				}
				add_to_variable = { TNO_tips_and_tricks_dirty = 0.001 }
			}
		}
	}
}
