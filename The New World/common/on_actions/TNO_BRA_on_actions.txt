on_actions = {
	on_daily_BRA = {
		effect = {
			if = {
				limit = { 
					has_country_flag = SACW_nukes 
					check_variable = { SA_nuclear_progress < 100 }
				}
				if = {
					limit = { check_variable = { SACW_nuclear_budget_days_remaining > 0 } } 
					add_to_variable = { SACW_nuclear_budget_days_remaining = -1 }
				}
				else = {
					set_country_flag = TNO_SACW_nuclear_budget_select_pop_up
					add_to_variable = { TNO_SACW_Select_Nuclear_Budget_dirty = 1 }
				}
			}
		}
	}
	on_weekly_BRA = {
		effect = {
			TNO_alert_suppression_weekly_tick = yes
			BRA_military_update = yes
		}
	}
	on_monthly_BRA = {
		effect = {
			log = "[GetDateText]: [Root.GetName]: LATIN SUBMOD: Brazil development graph pulse"
			BRA_monthly_on_action_effect = yes
			if = {
				limit = {
					has_idea = BRA_lacerda_paulista_riots
				}
				BRA_lacerda_liberdade_on_fire_monthly = yes
			}
			if = {
				limit = {
					has_idea = BRA_Liberdade_in_Ashes
				}
				BRA_lacerda_liberdade_in_ashes_monthly = yes
			}
			if = {
				limit = {
					OR = {
						has_country_flag = BRA_sorbonne_lacerda #sorbonne
						has_country_flag = BRA_lacerda_coup_happened #imperial
						has_country_flag = BRA_65_lacerda_presidency #regulerial
					}
				}
				set_temp_variable = { y = BRA_lacerda_udn_manage_econ_tick }
				divide_temp_variable = { y = 10 }
				BRA_UDN_management_change_y = yes
				set_temp_variable = { x = BRA_lacerda_udn_manage_pol_tick }
				divide_temp_variable = { x = 10 }
				BRA_UDN_management_change_x = yes
				if = {
					limit = { BRA_lacerda_udn_party_balanced = yes }
					set_temp_variable = { temp_udn_unity = 1 }
					BRA_lacerda_udn_unity_change = yes
				}
				else_if = {
					limit = { BRA_lacerda_udn_banda_quarter = yes }
					set_temp_variable = { temp_udn_unity = 1 }
					BRA_lacerda_udn_unity_change = yes
				}
				else_if = {
					limit = { BRA_lacerda_udn_bossa_quarter = yes }
					set_temp_variable = { temp_udn_unity = -2 }
					BRA_lacerda_udn_unity_change = yes
				}
				else_if = {
					limit = { BRA_lacerda_udn_realistas_quarter = yes }
					set_temp_variable = { temp_udn_unity = -1 }
					BRA_lacerda_udn_unity_change = yes
				}
				else_if = {
					limit = { BRA_lacerda_udn_liberais_quarter = yes }
					set_temp_variable = { temp_udn_unity = 1 }
					BRA_lacerda_udn_unity_change = yes
				}
			}
		}
	}
	on_declare_war = {
		effect = {
			BRA = {
				if = {
					limit = {
						NOT = { has_country_flag = BRA_SAW_event }
						ROOT = {
							OR = {
								tag = SAF
								tag = ANG
								tag = GRO
								tag = COG
							}
						}
						FROM = {
							OR = {
								tag = SAF
								tag = ANG
								tag = GRO
								tag = COG
							}
						}
					}
					set_country_flag = BRA_SAW_event
					country_event = { id = brazil.7 days = 6 }
				}
				if = {
					limit = {
						NOT = { has_country_flag = BRA_GCW_event }
						ROOT = {
							OR = {
								tag = SGR
								tag = GGR
								tag = BGR
								tag = HGR
							}
						}
						FROM = {
							OR = {
								tag = SGR
								tag = GGR
								tag = BGR
								tag = HGR
							}
						}
					}
					set_country_flag = BRA_GCW_event
					country_event = { id = brazil.265 days = 21 }
				}
			}
		}
	}
	on_capitulation = {
		effect = {
			BRA = {
				if = {
					limit = {
						NOT = { has_country_flag = BRA_SAW_ends_event }
						FROM = {
							OR = {
								tag = SAF
								tag = ANG
								tag = GRO
								tag = COG
							}
						}
						ROOT = {
							OR = {
								tag = SAF
								tag = ANG
								tag = GRO
								tag = COG
							}
							OR = {
								is_in_faction = no
								all_allied_country = {
									has_capitulated = yes
								}
							}
						}
					}
					set_country_flag = BRA_SAW_ends_event
					country_event = { id = brazil.110 days = 1 }
				}
			}
		}
	}
}
