#Economic Factor Score
@SACW_score_per_gdp = 1.2   # Num. of points given by each 1B in GDP
@SACW_score_per_gdpc = 1     # Num. of points given by each K in GDP/Capita
@SACW_score_per_debt = 0.3   # Num. of points subtracted by each 1B in debt
@SACW_score_per_poverty = 0.3   # Num. of points subtracted by each % of poverty rate
@SACW_score_per_stability = 0.5   # Num. of points given by each % in stability

## Military Score ##
@sacw_score_per_battalion      = 0.15  # Num. of points given by each fielded battalion

## Naval Score ##
@sacw_score_per_submarine      = 0.4   # Num. of points given by each submarine
@sacw_score_per_corvette       = 0.5   # Num. of points given by each corvette
@sacw_score_per_destroyer      = 0.75  # Num. of points given by each destroyer
@sacw_score_per_frigate        = 1     # Num. of points given by each frigate
@sacw_score_per_cruiser        = 2   # Num. of points given by each cruiser
@sacw_score_per_battleship     = 4  # Num. of points given by each battleship
@sacw_score_per_carrier        = 4     # Num. of points given by each carrier

## Air Force Score ##

@sacw_score_per_heavy = 0.1
@sacw_score_per_small = 0.05

sacw_calculate_army_power = {
	set_variable = { sacw_army_power = num_battalions }
	multiply_variable = { sacw_army_power = @sacw_score_per_battalion }
}

sacw_calculate_naval_power = {

	set_variable = { SACWSubmarineScore = num_ships_with_type@submarine }
	multiply_variable = { SACWSubmarineScore = @sacw_score_per_submarine }
	set_variable = { SACWCorvetteScore = num_ships_with_type@corvette }
	multiply_variable = { SACWCorvetteScore = @sacw_score_per_corvette }
	set_variable = { SACWDestroyerScore = num_ships_with_type@destroyer }
	multiply_variable = { SACWDestroyerScore = @sacw_score_per_destroyer }
	set_variable = { SACWFrigateScore = num_ships_with_type@frigate }
	multiply_variable = { SACWFrigateScore = @sacw_score_per_frigate }
	set_variable = { SACWCruiserScore = num_ships_with_type@cruiser }
	multiply_variable = { SACWCruiserScore = @sacw_score_per_cruiser }
	set_variable = { SACWBattleshipScore = num_ships_with_type@battleship }
	multiply_variable = { SACWBattleshipScore = @sacw_score_per_battleship }
	set_variable = { SACWCarrierScore = num_ships_with_type@carrier }
	multiply_variable = { SACWCarrierScore = @sacw_score_per_carrier }

	set_variable = { sacw_naval_power = 0 }

	add_to_variable = { sacw_naval_power = SACWSubmarineScore }
	add_to_variable = { sacw_naval_power = SACWCorvetteScore }
	add_to_variable = { sacw_naval_power = SACWDestroyerScore }
	add_to_variable = { sacw_naval_power = SACWFrigateScore }
	add_to_variable = { sacw_naval_power = SACWCruiserScore }
	add_to_variable = { sacw_naval_power = SACWBattleshipScore }
	add_to_variable = { sacw_naval_power = SACWCarrierScore }

}

SA_cold_war_score = {
	#Political

	set_variable = { SA_cws_political = 0 }
	clear_array = SA_cws_political_allies
	clear_array = SA_cws_political_allies_points

	for_each_loop = {
		array = global.TNO_SA_cold_war_nations

		set_temp_variable = { nation = global.TNO_SA_cold_war_nations^i }
		if = {
			limit = { check_variable = { nation:SA_cold_war_alignment = tempCheck } }
			add_to_variable = { SA_cws_political = nation:SA_cold_war_value }
			add_to_array = { SA_cws_political_allies = nation }
			add_to_array = { SA_cws_political_allies_points = nation:SA_cold_war_value }
		}
	}

	# Sort the arrays from highest number of points to lowest
	for_loop_effect = {
		start = 0
		end = SA_cws_political_allies^num
		value = base

		set_temp_variable = { lowest = 999999 }
		set_temp_variable = { tgt = -1 }

		for_loop_effect = {
			start = base
			end = SA_cws_political_allies^num
			value = i

			if = {
				limit = { check_variable = { SA_cws_political_allies_points^i < lowest } }

				set_temp_variable = { lowest = SA_cws_political_allies_points^i }
				set_temp_variable = { tgt = i }
			}
		}

		if = {
			limit = { check_variable = { tgt > -1 } }

			set_temp_variable = { tmp_identity = SA_cws_political_allies^tgt }
			set_temp_variable = { tmp_points = SA_cws_political_allies_points^tgt }

			remove_from_array = { array = SA_cws_political_allies index = tgt }
			remove_from_array = { array = SA_cws_political_allies_points index = tgt }

			add_to_array = { array = SA_cws_political_allies value = tmp_identity index = 0 }
			add_to_array = { array = SA_cws_political_allies_points value = tmp_points index = 0 }

		}
	}

	#Economy

	# Base array
	clear_array = SA_cws_economy
	add_to_array = { SA_cws_economy = 0 } # 0; Total
	add_to_array = { SA_cws_economy = 0 } # 1; GDP/Capita
	add_to_array = { SA_cws_economy = 0 } # 2; National Debt
	add_to_array = { SA_cws_economy = 0 } # 3; Poverty Rate
	add_to_array = { SA_cws_economy = 0 } # 4; Stability

	# GDP/Capita
	set_variable = { SA_cws_economy^1 = GDP }
	set_variable = { SA_country_pop = max_manpower_k }
	divide_variable = { SA_country_pop = 100000 }
	divide_variable = { SA_cws_economy^1 = SA_country_pop }
	multiply_variable = { SA_cws_economy^1 = @SACW_score_per_gdpc }

	# National Debt
	set_variable = { SA_cws_economy^2 = national_debt }
	multiply_variable = { SA_cws_economy^2 = @SACW_score_per_debt }
	multiply_variable = { SA_cws_economy^2 = -0.5 }

	# Poverty Rate
	set_variable = { SA_cws_economy^3 = poverty_rate }
	multiply_variable = { SA_cws_economy^3 = @SACW_score_per_poverty }
	multiply_variable = { SA_cws_economy^3 = -0.5 }

	# Stability
	set_variable = { SA_cws_economy^4 = stability }
	multiply_variable = { SA_cws_economy^4 = 100 }
	multiply_variable = { SA_cws_economy^4 = @SACW_score_per_stability }
	clamp_variable = { var = SA_cws_economy^4 min = -50 max = 50 }

	# Rounding up all variables
	for_each_loop = {
		array = SA_cws_economy
		round_variable = SA_cws_economy^i
		add_to_variable = { SA_cws_economy^0 = SA_cws_economy^i }
	}


	#Military
	clear_array = SA_cws_military
	add_to_array = { SA_cws_military = 0 } # 0; Total
	add_to_array = { SA_cws_military = 0 } # 1; Army Size
	add_to_array = { SA_cws_military = 0 } # 2; Navy Size
	add_to_array = { SA_cws_military = 0 } # 3; Air Force Size

	sacw_calculate_army_power = yes
	set_variable = { SA_cws_military^1 = sacw_army_power }

	# Naval Size score
	sacw_calculate_naval_power = yes
	set_variable = { SA_cws_military^2 = sacw_naval_power }

	# Air Force Size score
	sacw_calculate_air_power = yes
	set_variable = { SA_cws_military^3 = sacw_air_force_power }

	for_each_loop = {
		array = SA_cws_military
		round_variable = SA_cws_military^i
		add_to_variable = { SA_cws_military^0 = SA_cws_military^i }
	}

	if = {
		limit = {
			BRA = {
				has_country_flag = public_declaration_nuke
			}
		}
		add_to_variable = {
			SA_cws_military^0 = 25
		}
	}

	if = {
		limit = {
			BRA = {
				has_country_flag = no_declaration_nuke
			}
		}
		add_to_variable = {
			SA_cws_military^0 = 5
		}
	}

	#Total
	set_variable = { SA_cws_total = 0 }
	add_to_variable = { SA_cws_total = SA_cws_economy^0 }
	add_to_variable = { SA_cws_total = SA_cws_political }
	add_to_variable = { SA_cws_total = SA_cws_military^0 }

}

#Adapted from Scars' influence battle script
SA_change_influence = {
	set_temp_variable = { v_check = TNO_SA_influence^tempFaction }
	add_to_temp_variable = { v_check = tempInfluence }

	if = {
		limit = {
			check_variable = { v_check < 0 }
		}
		subtract_from_temp_variable = { tempInfluence = v_check }
	}
	else_if = {
		limit = {
			check_variable = { v_check > 100 }
		}
		set_temp_variable = { n_100 = v_check }
		subtract_from_temp_variable = { n_100 = 100 }
		subtract_from_temp_variable = { tempInfluence = n_100 }
	}

	if = {
		limit = {
			NOT = {
				check_variable = { tempInfluence = 0 }
			}
		}
		custom_effect_tooltip = SA_influence_change_tt

		add_to_variable = { TNO_SA_influence^tempFaction = tempInfluence }

		for_each_loop = {
			array = TNO_SA_influence
			if = {
				limit = {
					check_variable = { v = 0 }
					NOT = { check_variable = { i = tempFaction } }
				}
				add_to_temp_array = { has_0 = i }
			}
		}

		set_temp_variable = { divide_factor = 6 }
		subtract_from_temp_variable = { divide_factor = has_0^num }
		divide_temp_variable = { tempInfluence = divide_factor }

		set_temp_variable = { excess = 0 }
		for_each_loop = {
			array = TNO_SA_influence
			if = {
				limit = {
					NOT = {
						check_variable = { i = tempFaction }
						is_in_array = { array = has_0 value = i }
					}
				}
				subtract_from_variable = { TNO_SA_influence^i = tempInfluence }
			}
			if = {
				limit = { check_variable = { TNO_SA_influence^i < 0 } }
				add_to_temp_variable = { excess = v }
				set_variable = { TNO_SA_influence^i = 0 }
				add_to_temp_array = { zero_value = i }
			}
		}
		for_each_loop = {
			array = TNO_SA_influence
			if = {
				limit = {
					NOT = {
						is_in_array = { array = zero_value value = i }
						check_variable = { i = tempFaction }
					}
				}
				add_to_variable = { TNO_SA_influence^i = excess }
			}
		}
	}

	#make sure it always adds to 100. Add unaligned if necessary
	set_temp_variable = { tempTotal = 0 }
	for_each_loop = {
		array = TNO_SA_influence
		add_to_temp_variable = { tempTotal = v }
	}
	set_temp_variable = { tempUnaligned = 100 }
	subtract_from_temp_variable = { tempUnaligned = tempTotal }
	add_to_variable = { TNO_SA_influence^0 = tempUnaligned }

	SA_set_alignment = yes
	SA_calculate_balance_of_power = yes
}

SA_set_alignment = {
	for_each_scope_loop = {
		array = global.TNO_SA_cold_war_nations

		set_temp_variable = { max_influence = 0 }
		set_temp_variable = { max_influence_faction = 0 }

		for_each_loop = {
			array = TNO_SA_influence
			index = influence_idx
			value = influence_value


			if = {
				limit = {
					AND = {
						check_variable = { influence_value > max_influence }
						NOT = { check_variable = { influence_value < 50 } }
					}
				}
				set_temp_variable = { max_influence = influence_value }
				set_temp_variable = { max_influence_faction = influence_idx }
			}
		}

		if = { # Brazil
			limit = { check_variable = { max_influence_faction = global.BRAZIL } }
			set_variable = { SA_cold_war_alignment = 2 }
		}
		else_if = { # argentina
			limit = { check_variable = { max_influence_faction = global.ARGENTINA } }
			set_variable = { SA_cold_war_alignment = 3 }
		}
		else_if = { # MEXICO
			limit = { check_variable = { max_influence_faction = global.MEXICO } }
			set_variable = { SA_cold_war_alignment = 4 }
		}
		else_if = { # Neutral bloc
			limit = { check_variable = { max_influence_faction = global.UNALIGNED } }
			set_variable = { SA_cold_war_alignment = 1 }
		}
		else_if = { # USA
			limit = { check_variable = { max_influence_faction = global.USA } }
			set_variable = { SA_cold_war_alignment = 6 }
		}
		else_if = { # Iberia
			limit = { check_variable = { max_influence_faction = global.IBERIA } }
			set_variable = { SA_cold_war_alignment = 5 }
		}
		else = { # Unaligned
			set_variable = { SA_cold_war_alignment = 1 }
		}

		set_variable = { SA_cold_war_display_frame = SA_cold_war_alignment }

		if = {
			limit = {
				OR = {
					tag = BRA
					tag = ARG
					tag = MEX
					tag = USA
				}
			}
			set_variable = { SA_cold_war_display_frame = 0 }
		}
	}
}

SATW_Escalation_Change = {
	set_temp_variable = { amount = global.SATW_Escalation_temp }
	if = {
		limit = { check_variable = { global.SATW_Escalation_temp < 0 } }
		multiply_temp_variable = { amount = -1 }
		custom_effect_tooltip = SATW_escalation_decrease_tt
	}
	else = {
		custom_effect_tooltip = SATW_escalation_increase_tt
	}
	add_to_variable = { global.SATW_Escalation = global.SATW_Escalation_temp }
	if = {
		limit = {
			has_game_rule = {
				rule = BRA_ARG_WAR_OUTCOME
				option = BRA_ARG_WAR_OUTCOME_STALEMATE
			}
		}
		set_variable = { global.SATW_Escalation = 0 }
	} else = {
		clamp_variable = {
			var = global.SATW_Escalation
			min = 0
			max = 100
		}
	}
	if = {
		limit = { check_variable = { global.SATW_Escalation = 0 } }
		set_global_flag = SATW_Stalemate
		TNO_SATW_end_war = yes
	}
	else_if = {
		limit = { check_variable = { global.SATW_Escalation = 100 } }
		BRA = { activate_mission = TNO_SATW_ww3_countdown }
	}
	hidden_effect = { SATW_Escalation_effect_update = yes }
}
