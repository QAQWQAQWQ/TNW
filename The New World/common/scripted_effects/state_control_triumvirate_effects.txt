state_control_effect_459 = {
	if = {
		limit = {
			OR = {
				is_controlled_by = IBR
				AND = {
					is_controlled_by = SLG
					SLG = {
						NOT= { has_cosmetic_tag = SLG_FREE_YVES }
					}
				}
			}
			has_global_flag = TNO_endonym_mode
		}
		set_state_name = STATE_459_spa
		set_province_name = { id = 7132 name = VICTORY_POINTS_7132_ita_spa }
		set_province_name = { id = 18075 name = VICTORY_POINTS_18075_spa }
	}
	else_if = {
		limit = {
			controller = { is_in_array = { global.italian_names = THIS } }
			has_global_flag = TNO_endonym_mode
		}
		set_province_name = { id = 7132 name = VICTORY_POINTS_7132_ita_spa }
	}
	else_if = {
		limit = { controller = { is_in_array = { global.arabic_names = THIS } } }
		set_state_name = STATE_459_alg
		if = {
			limit = { has_global_flag = TNO_endonym_mode }
			set_province_name = { id = 4150 name = VICTORY_POINTS_4150_alg }
			set_province_name = { id = 7132 name = VICTORY_POINTS_7132_alg }
			set_province_name = { id = 18075 name = VICTORY_POINTS_18075_alg }
		}
	}
	else = {
		reset_state_name = yes
		reset_province_name = 4150
		reset_province_name = 7132
		reset_province_name = 18075
	}
}
