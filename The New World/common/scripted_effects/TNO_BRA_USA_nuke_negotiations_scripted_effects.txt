BRA_nuke_negotiations_end_negotiations = {
	custom_effect_tooltip = BRA_nuke_negotiations_end_neg_tt
	TNO_fopo_remove_nuke_negotiations_tab = yes
	if = {
		limit = { tag = BRA }
		set_temp_variable = { temp_bill = token:BRA_BRAZIL_AMERICA_ARMS_TREATY }
		set_variable = { BRA_pl_ptb_support_bill@var:temp_bill = 0.2 }
		set_variable = { BRA_pl_psd_support_bill@var:temp_bill = 0.2 }
		BRA_lacerda_calculate_udn_bill_support = yes
		set_variable = { BRA_pl_psp_support_bill@var:temp_bill = 0.6 }
		if = {
			limit = { has_country_flag = BRA_lacerda_psp_support_flag }
			if = {
				limit = { check_variable = { BRA_pl_psp_support_bill@var:temp_bill < BRA_lacerda_psp_support } }
				set_variable = { BRA_pl_psp_support_bill@var:temp_bill = BRA_lacerda_psp_support }
			}
			if = {
				limit = { check_variable = { BRA_lacerda_psp_support = 0 } }
				set_variable = { BRA_pl_psp_support_bill@var:temp_bill = BRA_lacerda_psp_support }
			}
		}
		set_variable = { BRA_pl_prp_support_bill@var:temp_bill = 0.6 }
		BRA_initialize_bill = yes
		set_variable = { BRA_pl_vote_days@var:temp_bill = 14 }
		BRA_call_bill_vote = yes

		country_event = {
			id = brazil_lacerda.73
			days = 10
		}
	}
	else_if = {
		limit = { tag = USA }
		country_event = brazil_lacerda.74
	}

	if = {
		limit = {
			check_variable = {
				TNO_nuke_negotiations_bias_BRA > 19
			}
		}
		set_variable = { TNO_fopo_nuke_neg_treaty_bias = 1 }
	}
	else_if = {
		limit = {
			check_variable = {
				TNO_nuke_negotiations_bias_BRA < 40
				TNO_nuke_negotiations_bias_BRA > 19
			}
		}
		set_variable = { TNO_fopo_nuke_neg_treaty_bias = 2 }
	}
	else_if = {
		limit = {
			check_variable = {
				TNO_nuke_negotiations_bias_BRA < 60
				TNO_nuke_negotiations_bias_BRA > 39
			}
		}
		set_variable = { TNO_fopo_nuke_neg_treaty_bias = 3 }
	}
	else_if = {
		limit = {
			check_variable = {
				TNO_nuke_negotiations_bias_BRA < 80
				TNO_nuke_negotiations_bias_BRA > 59
			}
		}
		set_variable = { TNO_fopo_nuke_neg_treaty_bias = 4 }
	}
	else_if = {

TNO_nuke_neg_succeed_chance_change = {
	set_temp_variable = { amount = TNO_nuke_negotiations_succeed_chance_temp }
	if = {
		limit = { check_variable = { TNO_nuke_negotiations_succeed_chance_temp < 0 } }
		multiply_temp_variable = { amount = -1 }
		custom_effect_tooltip = TNO_nuke_neg_succeed_change_decrease_tt
	}
	else = {
		custom_effect_tooltip = TNO_nuke_neg_succeed_change_increase_tt
	}
	add_to_variable = { TNO_nuke_negotiations_succeed_chance = TNO_nuke_negotiations_succeed_chance_temp }
	subtract_from_variable = { TNO_nuke_negotiations_fail_chance = TNO_nuke_negotiations_succeed_chance_temp }
	clamp_variable = {
		var = TNO_nuke_negotiations_succeed_chance
		min = 0
		max = 100
	}
	clamp_variable = {
		var = TNO_nuke_negotiations_fail_chance
		min = 0
		max = 100
	}
}
