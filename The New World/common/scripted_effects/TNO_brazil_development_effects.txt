BRA_SetEconomicGoals = {
	if = {
		limit = { check_variable = { global.year = 1962 } }
		set_variable = { BRA_TargetGDPGrowth = 33.785 }
		set_variable = { BRA_TargetInflation = 28.3 }
		set_variable = { BRA_last_year_gdp = 32.06 }
	}
	else = {
		# Calculate growth since last year
		set_temp_variable = { year_on_year_growth = BRA.GDP }
		divide_temp_variable = { year_on_year_growth = BRA_last_year_gdp }

		set_variable = { BRA_last_year_gdp = BRA.GDP }

		# Calculating our GDP growth target
		set_temp_variable = { BRA_growth_target_as_mult = year_on_year_growth }

		# Get average between our calculated target and a +10% target to avoid extreme targets.
		add_to_temp_variable = { BRA_growth_target_as_mult = 1.1 }
		divide_temp_variable = { BRA_growth_target_as_mult = 2 }

		add_to_temp_variable = { BRA_growth_target_as_mult = BRA_gdp_target_additive }

		set_variable = { BRA_TargetGDPGrowth = BRA.GDP }
		multiply_variable = { BRA_TargetGDPGrowth = BRA_growth_target_as_mult }

		# Set our inflation target
		set_variable = { BRA_TargetInflation = BRA.clamped_inflation_rate }
		subtract_from_variable = { BRA_TargetInflation = BRA.rubberband_inflation_rate }
		multiply_variable = { BRA_TargetInflation = 0.9 }
		add_to_variable = { BRA_TargetInflation = BRA.rubberband_inflation_rate }
	}
}

BRA_check_TargetGDPGrowth_success = {
	# No check when it is just 3 months after the presidential election
	if = {
		limit = {
			OR = {
				check_variable = { global.year = 1966 }
				check_variable = { global.year = 1971 }
			}
		}
		country_event = { id = brazil.1005 days = 7 }
	}
	else = {
		set_variable = { bra_econ_successes = 0 }
		if = {
			limit = { check_variable = { BRA.GDP > BRA_TargetGDPGrowth } }
			add_to_variable = { bra_econ_successes = 1 }
			add_to_variable = { BRA_gdp_target_additive = 0.01 }
		}
		else = {
			add_to_variable = { BRA_gdp_target_additive = -0.02 }
		}
		if = {
			limit = { check_variable = { clamped_inflation_rate < BRA_TargetInflation } }
			add_to_variable = { bra_econ_successes = 1 }
		}

		if = {
			limit = { check_variable = { bra_econ_successes = 2 } }
			country_event = brazil.1002
		}
		else_if = {
			limit = { check_variable = { bra_econ_successes = 1 } }
			country_event = brazil.1003
		}
		else = {
			country_event = brazil.1004
		}
	}
	BRA_SetEconomicGoals = yes
}

BRA_develop_initialize_amazonas = {
	clear_array = BRA_develop_amazonas_array
	set_variable = { BRA_develop_rainforest_cover = 0 }
	set_variable = { BRA_develop_rainforest_cover_percentage = 1 }
	set_variable = { BRA_develop_forest_loss = 2 }
	set_variable = { BRA_develop_regional_progress^0 = 4 }
	set_variable = { BRA_develop_regional_progress_amazonia = BRA_develop_regional_progress^0 }

	add_to_array = { BRA_develop_amazonas_array = 4018.001 }
	set_temp_variable = { x = BRA_develop_amazonas_array^BRA_develop_rainforest_cover }
	multiply_temp_variable = { x = 100 }
	divide_temp_variable = { x = 4100.001 }
	add_to_array = { BRA_develop_amazonas_array = x }
	add_to_array = { BRA_develop_amazonas_array = 1500 }

	if = {
		limit = {
			check_variable = {
				BRA_megaprojects_progress^TRANS_AMAZONIAN_HIGHWAY_MEGAPROJECT = 100
			}
		}
		add_to_variable = { BRA_develop_amazonas_array^BRA_develop_forest_loss = 1000 }
	}
}

BRA_develop_amazonas_yearly = {
	set_temp_variable = { x = BRA_develop_amazonas_array^BRA_develop_forest_loss }
	divide_temp_variable = { x = 1000 }
	subtract_from_variable = { BRA_develop_amazonas_array^BRA_develop_rainforest_cover = x }
	set_temp_variable = { x = BRA_develop_amazonas_array^BRA_develop_rainforest_cover }
	multiply_temp_variable = { x = 100 }
	divide_temp_variable = { x = 4100.001 }
	set_variable = { BRA_develop_amazonas_array^BRA_develop_rainforest_cover_percentage = x }
	subtract_from_temp_variable = { x = 100 } # Amazonia development goes from 5 -> 1 over time, the reverse of other regions
	divide_temp_variable = { x = 5 }
	add_to_temp_variable = { x = 4.5 }
	round_temp_variable = x
	clamp_temp_variable = {
		var = x
		min = 1
		max = 5
	}
	set_variable = { BRA_develop_regional_progress_amazonia = x }
	set_variable = { BRA_develop_regional_progress^0 = x }
	BRA_deforestation_events = yes
}

BRA_clamp_deforestation = {
	clamp_temp_variable = {
		var = BRA_develop_amazonas_array^BRA_develop_forest_loss
		min = 0
		max = 40000
	}
}

BRA_deforestation_events = {
	if = {
		limit = {
			check_variable = { BRA_develop_amazonas_array^BRA_develop_rainforest_cover_percentage < 97.01 }
		}
		country_event = BRA_deforestation.1
	}
	if = { #not else if or the events might be overran
		limit = {
			check_variable = { BRA_develop_amazonas_array^BRA_develop_rainforest_cover_percentage < 96.01 }
		}
		country_event = BRA_deforestation.2
	}
	if = {
		limit = {
			check_variable = { BRA_develop_amazonas_array^BRA_develop_rainforest_cover_percentage < 95.01 }
		}
		country_event = BRA_deforestation.3
	}
	if = {
		limit = {
			check_variable = { BRA_develop_amazonas_array^BRA_develop_rainforest_cover_percentage < 93.01 }
		}
		country_event = BRA_deforestation.4
	}
	if = {
		limit = {
			check_variable = { BRA_develop_amazonas_array^BRA_develop_rainforest_cover_percentage < 90.01 }

		country_event = BRA_deforestation.5
	}
}
