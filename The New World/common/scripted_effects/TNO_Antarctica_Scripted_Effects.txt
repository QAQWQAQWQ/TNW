# DESCRIPTION: Initializes whaling nat spirit. Can be used to add another benificiary. If banned will remove idea
# INPUT: tempChange, tempTag
Antarctica_Initialize_Whaling = {
	var:tempTag = {
		add_dynamic_modifier = { modifier = Antarctica_Whaling_Income }
	}
	add_to_array = { Antarctica_WhalingBenificiary = tempTag }

	# Japan have their own spirit 
	if = {
		limit = { NOT = { tag = AAJ } }
		meta_effect = {
			text = {
				remove_ideas = Antarctica_whaling_banned_[TAG]
				add_ideas = Antarctica_whaling_[TAG]
			}
			TAG = "[THIS.GetTag]"
		}
	}
	add_to_array = { Antarctica_WhalingIncome_M = 0 }
	Antarctica_Change_WhalingIncome = yes
}

# DESCRIPTION: Changes whaling income for tempTag
# INPUT: tempChange, tempTag
Antarctica_Change_WhalingIncome = {
	for_each_loop = {
		array = Antarctica_WhalingBenificiary
		if = {
			limit = { check_variable = { tempTag = Antarctica_WhalingBenificiary^i } }
			set_temp_variable = { idx = i }
		}
	}

	var:tempTag = {
		add_to_variable = { Antarctica_WhalingIncome = tempChange }

		# Converts value into millions for display on antarctica tag
		set_variable = { PREV.Antarctica_WhalingIncome_M^idx = THIS.Antarctica_WhalingIncome }
		multiply_variable = { PREV.Antarctica_WhalingIncome_M^idx = 1000 }
	}

	# Correct units for tooltip
	multiply_temp_variable = { tempChange = 1000 }
	if = {
		limit = { check_variable = { tempChange > 0 } }
		custom_effect_tooltip = Antarctica_Change_WhalingIncome_increase_tt
	}
	else = {
		multiply_temp_variable = { tempChange = -1 }
		custom_effect_tooltip = Antarctica_Change_WhalingIncome_decrease_tt
	}

	force_update_dynamic_modifier = yes
}

# DESCRIPTION: Bans whaling and removes overlord income, defaults to Antarctica_WhalingBenificiary
# INPUT: N/A
Antarctica_Ban_Whaling = {
	for_each_loop = {
		array = Antarctica_WhalingBenificiary
		var:Antarctica_WhalingBenificiary^i = {
			remove_dynamic_modifier = { modifier = Antarctica_Whaling_Income }
			clear_variable = Antarctica_WhalingIncome
		}
	}
	clear_array = Antarctica_WhalingBenificiary
	
	meta_effect = {
		text = {
			remove_ideas = {
				Antarctica_whaling_[TAG]
				Antarctica_whaling_booming_[TAG]
			}
			add_ideas = Antarctica_whaling_banned_[TAG]
		}
		TAG = "[THIS.GetTag]"
	}
}

Antarctica_AAO_Successor_Update_Pie_Chart = {
	# Clamp
	clamp_variable = {
		var = Antarctica_GUI_AAO_USA_Control_Percent
		max = 100
		min = 0
	}
	round_variable = Antarctica_GUI_AAO_USA_Control_Percent
	clamp_variable = {
		var = Antarctica_GUI_AAO_AST_Control_Percent
		max = 100
		min = 0
	}
	round_variable = Antarctica_GUI_AAO_AST_Control_Percent
	clamp_variable = {
		var = Antarctica_GUI_AAO_NZL_Control_Percent
		max = 100
		min = 0
	}
	round_variable = Antarctica_GUI_AAO_NZL_Control_Percent
	
	# Overflow Check
	set_temp_variable = { sum = Antarctica_GUI_AAO_USA_Control_Percent }
	add_to_temp_variable = { sum = Antarctica_GUI_AAO_AST_Control_Percent }
	add_to_temp_variable = { sum = Antarctica_GUI_AAO_NZL_Control_Percent }
	if = {
		limit = {
			check_variable = { sum > 100 }
		}
		set_temp_variable = { excess = sum }
		add_to_temp_variable = { excess = -100 }
		if = {
			limit = {
				check_variable = { Antarctica_GUI_AAO_USA_Control_Percent > excess }
			}
			subtract_from_variable = { Antarctica_GUI_AAO_USA_Control_Percent = excess }
		}
		else_if = {
			limit = {
				check_variable = { Antarctica_GUI_AAO_AST_Control_Percent > excess }
			}
			subtract_from_variable = { Antarctica_GUI_AAO_AST_Control_Percent = excess }
		}
		else_if = {
			limit = {
				check_variable = { Antarctica_GUI_AAO_NZL_Control_Percent > excess }
			}
			subtract_from_variable = { Antarctica_GUI_AAO_NZL_Control_Percent = excess }
		}
	}
	if = {
		limit = {
			check_variable = { sum < 100 }
		}
		set_temp_variable = { deficit = 100 }
		subtract_from_temp_variable = { deficit = sum }
		if = {
			limit = {
				check_variable = { Antarctica_GUI_AAO_USA_Control_Percent < deficit }
			}
			add_to_variable = { Antarctica_GUI_AAO_USA_Control_Percent = deficit }
		}
		else_if = {
			limit = {
				check_variable = { Antarctica_GUI_AAO_AST_Control_Percent < deficit }
			}
			add_to_variable = { Antarctica_GUI_AAO_AST_Control_Percent = deficit }
		}
		else_if = {
			limit = {
				check_variable = { Antarctica_GUI_AAO_NZL_Control_Percent < deficit }
			}
			add_to_variable = { Antarctica_GUI_AAO_NZL_Control_Percent = deficit }
		}
		else = {
			random_list = {
				1 = {
					add_to_variable = { Antarctica_GUI_AAO_USA_Control_Percent = deficit }
				}
				1 = {
					add_to_variable = { Antarctica_GUI_AAO_AST_Control_Percent = deficit }
				}
				1 = {
					add_to_variable = { Antarctica_GUI_AAO_NZL_Control_Percent = deficit }
				}
			}
		}
	}
	
	# Pie Chart update
	set_variable = { Antarctica_GUI_AAO_USA_Control = Antarctica_GUI_AAO_USA_Control_Percent }

	set_variable = { Antarctica_GUI_AAO_AST_Control = Antarctica_GUI_AAO_AST_Control_Percent }
	add_to_variable = { Antarctica_GUI_AAO_AST_Control = Antarctica_GUI_AAO_USA_Control }

	set_variable = { Antarctica_GUI_AAO_NZL_Control = Antarctica_GUI_AAO_NZL_Control_Percent }
	add_to_variable = { Antarctica_GUI_AAO_NZL_Control = Antarctica_GUI_AAO_AST_Control }

	set_popularities = {
		paternalism_1 = Antarctica_GUI_AAO_AST_Control_Percent
		paternalism = Antarctica_GUI_AAO_USA_Control_Percent
		paternalism_2 = Antarctica_GUI_AAO_NZL_Control_Percent
	}
}
