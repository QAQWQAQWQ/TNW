TNO_iwr_pandoras_box_change_days = {
	if = {
		limit = {
			check_variable = { days_temp < 0 }
		}
		effect_tooltip = { #For the tooltip to show positive 
			multiply_temp_variable = {
				var = days_temp
				value = -1
			}
		}
		custom_effect_tooltip = TNO_iwr_pandoras_box_remove_days_tt
		if = {
			limit = { original_tag = USA } # For debugging purposes
			log = "[GetDateText]: [Root.GetName]: Subtracted [?days_temp|0] days from [IND.GetName]'s Pandora's Box"
		}
		else = {
			log = "[GetDateText]: [Root.GetName]: Subtracted [?days_temp|0] days from [AZH.GetName]'s Pandora's Box"
		}
	}
	else = {
		custom_effect_tooltip = TNO_iwr_pandoras_box_add_days_tt
		if = {
			limit = { original_tag = USA }
			log = "[GetDateText]: [Root.GetName]: Added [?days_temp|0] days to [IND.GetName]'s Pandora's Box"
		}
		else = {
			log = "[GetDateText]: [Root.GetName]: Added [?days_temp|0] days to [AZH.GetName]'s Pandora's Box"
		}
	}
	hidden_effect = {
		if = {
			limit = { original_tag = USA }
			ZZZ = {
				meta_effect = { 
					text = {
						add_days_mission_timeout = {
							mission =  ZZZ_USA_IND_Pandoras_Box_Next_Stage
							days = [pb_days]
						}
					}
					pb_days = "[?days_temp|0]"
				}
			}
		}
		else = {
			ZZZ = {
				meta_effect = { 
					text = {
						add_days_mission_timeout = {
							mission =  ZZZ_JAP_AZH_Pandoras_Box_Next_Stage
							days = [pb_days]
						}
					}
					pb_days = "[?days_temp|0]"
				}
			}
		}
	}
}

TNO_iwr_light_military_aid_effect = {
	set_temp_variable = { new_inf_amt_temp = global.int_inf_amt }
	set_temp_variable = { new_sup_amt_temp = global.int_sup_amt }
	set_temp_variable = { new_mot_amt_temp = global.int_mot_amt }

	if = {
		limit = { original_tag = USA }
		set_temp_variable = { target_country = IND }
	}
	else = {
		set_temp_variable = { target_country = AZH }
	}
	
	if = {
		limit = {
			var:target_country = { check_variable = { pb_stage = 1 } }
		}
		set_temp_variable = { light_mil_aid_mult_temp = 1 }
	}
	else_if = {
		limit = {
			var:target_country = { check_variable = { pb_stage = 2 } }
		}
		set_temp_variable = { light_mil_aid_mult_temp = 2 }
	}
	else_if = {
		limit = {
			var:target_country = { check_variable = { pb_stage = 3 } }
		}
		set_temp_variable = { light_mil_aid_mult_temp = 3 }
	}
	else_if = {
		limit = {
			var:target_country = { check_variable = { pb_stage = 4 } }
		}
		set_temp_variable = { light_mil_aid_mult_temp = 4 }
	}
	else_if = {
		limit = {
			var:target_country = { check_variable = { pb_stage = 5 } }
		}
		set_temp_variable = { light_mil_aid_mult_temp = 5 }
	}

	multiply_temp_variable = { new_inf_amt_temp = light_mil_aid_mult_temp }
	multiply_temp_variable = { new_sup_amt_temp = light_mil_aid_mult_temp }
	multiply_temp_variable = { new_mot_amt_temp = light_mil_aid_mult_temp }

	send_equipment = {
		equipment = infantry_equipment
		amount = var:new_inf_amt_temp
		target = var:target_country
	}
	send_equipment = {
		equipment = support_equipment
		amount = var:new_sup_amt_temp
		target = var:target_country
	}
	send_equipment = {
		equipment = motorized_equipment
		amount = var:new_mot_amt_temp
		target = var:target_country
	}
}
