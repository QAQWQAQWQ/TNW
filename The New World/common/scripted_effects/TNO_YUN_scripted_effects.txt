		max = 20

WI_calc_admin_efficiency_rate = {
	if = {
		limit = { check_variable = { china_state_ref = i } }
		# Legitimacy
		set_variable = { ROOT.greater_china_province_admin_legitimacy = ROOT.YUN_legitimacy }
		multiply_variable = { ROOT.greater_china_province_admin_legitimacy = 0.1 }
		add_to_variable = { ROOT.greater_china_province_admin_rate^i = ROOT.greater_china_province_admin_legitimacy }
		# Provincal Capital
		if = {
			limit = {
				is_controlled_by = ROOT
			}
			multiply_variable = { ROOT.greater_china_province_admin_rate^i = 1.5 }
		}
		# National Spending
		set_variable = { ROOT.greater_china_province_admin_national = ROOT.econ_admin_expenditures_slider_pct }
		multiply_variable = { ROOT.greater_china_province_admin_national = 0.01 }
		multiply_variable = { ROOT.greater_china_province_admin_rate^i = ROOT.greater_china_province_admin_national }
		# Local Spending
		set_variable = { THIS.greater_china_province_admin_local = THIS.WI_admin_spending_percent }
		multiply_variable = { THIS.greater_china_province_admin_local = 10 }
		multiply_variable = { ROOT.greater_china_province_admin_rate^i = THIS.greater_china_province_admin_local }
		# Modifiers
		multiply_variable = { ROOT.greater_china_province_admin_rate^i = ROOT.greater_china_province_admin_bonus }
		# Base rate
		multiply_variable = { ROOT.greater_china_province_admin_rate^i = ROOT.greater_china_province_admin_base^i }
		set_variable = { greater_china_province_admin_rate_info = ROOT.greater_china_province_admin_rate^i }
	}
}
