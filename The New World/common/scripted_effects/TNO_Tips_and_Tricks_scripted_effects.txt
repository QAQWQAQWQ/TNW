TNO_Tips_and_Tricks_overflow_solution = {
	set_temp_variable = { temp = global.tip_category_array^num }
	subtract_from_temp_variable = { temp = 1 }
	if = {
		limit = {
			check_variable = { current_tip_id > temp }
		}
		set_variable = { current_tip_id = 0 }
	}
	else_if = {
		limit = {
			check_variable = { current_tip_id < 0 }
		}
		set_variable = { current_tip_id = temp }
	}
}
