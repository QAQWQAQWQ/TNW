Antarctica_AAC_Change_XP = {
	add_to_variable = { AAC_Skill_XP = temp_value }
	if = {
		limit = {
			check_variable = { temp_value > 0 }
		}
		custom_effect_tooltip = Antarctica_AAC_Add_XP_tt
	}
	else = {
		multiply_temp_variable = { temp_value = -1 }
		custom_effect_tooltip = Antarctica_AAC_Subtract_XP_tt
	}
	clamp_variable = { var = AAC_Skill_XP min = 0 max = 100 }
	add_to_variable = { dirty_marines = 1 }
}

Antarctica_AAC_Marine_Modify_Skill = {
	add_to_variable = { AAC_Skill_Array^temp_skill = temp_value }
	if = {
		limit = {
			check_variable = { temp_value > 0 }
		}
		custom_effect_tooltip = AAC_marine_increase_skill_tt
	}
	
	else = {
		multiply_temp_variable = { temp_value = -1 }
		custom_effect_tooltip = AAC_marine_decrease_skill_tt
	}
	clamp_variable = { var = AAC_Skill_Array^temp_skill min = 0 max = 100 }

	Antarctica_AAC_Marine_Skill_Check = yes
	add_to_variable = { dirty_marines = 1 }
}

Antarctica_AAC_Marine_Skill_Check = {
	# Mobility
	set_variable = { AAC_Mobility_dynmod_var = AAC_Skill_Array^0 }
	divide_variable = { AAC_Mobility_dynmod_var = 500 }

	set_variable = { AAC_Mobility_resource = AAC_Skill_Array^0 } # legitimacy
	divide_variable = { AAC_Mobility_resource = 100 }
	subtract_from_variable = { AAC_Mobility_resource = 0.5 }

	# Endurance
	set_variable = { AAC_Endurance_dynmod_var = AAC_Skill_Array^1 }
	divide_variable = { AAC_Endurance_dynmod_var = 200 }

	set_variable = { AAC_Endurance_resource = AAC_Skill_Array^1 } # supplies
	divide_variable = { AAC_Endurance_resource = -200 }
	add_to_variable = { AAC_Endurance_resource = 0.5 }

	# Aggression
	set_variable = { AAC_Aggression_dynmod_var = AAC_Skill_Array^2 }
	divide_variable = { AAC_Aggression_dynmod_var = 200 }

	set_variable = { AAC_Aggression_resource = AAC_Skill_Array^2 } # militarisation
	divide_variable = { AAC_Aggression_resource = 100 }
	subtract_from_variable = { AAC_Aggression_resource = 0.5 }

	# Loyalty
	set_variable = { AAC_Loyalty_dynmod_var = AAC_Skill_Array^3 }
	divide_variable = { AAC_Loyalty_dynmod_var = 200 }

	set_variable = { AAC_Loyalty_resource = AAC_Skill_Array^3 } # PP
	divide_variable = { AAC_Loyalty_resource = 200 }
	subtract_from_variable = { AAC_Loyalty_resource = 0.25 }

	force_update_dynamic_modifier = yes
}

# Used to display a tooltip fo changing equipment
Antarctica_AAC_Marine_Remove_Current_Equipment = { # fires first to remove current equipment

	set_temp_variable = { Equipment = AAC_Active_Equipment_Array^AAC_Marine_Active_Category }
	Antarctica_AAC_Marine_Assign_Equipment_Values = yes # look up the values for this equipment
	
	# apply the change
	subtract_from_variable = { AAC_Skill_Array^0 = mobility }
	subtract_from_variable = { AAC_Skill_Array^1 = endurance }
	subtract_from_variable = { AAC_Skill_Array^2 = aggression }
	subtract_from_variable = { AAC_Skill_Array^3 = loyalty }

	custom_effect_tooltip = Antarctica_AAC_Marine_Remove_Current_Equipment_Name_tt
	Antarctica_AAC_Marine_Equipment_Effect_Tooltip = yes #show a tooltip
}

# Used to unlock new equipment or display a tooltip for changing equipment
Antarctica_AAC_Marine_Add_Equipment = {
	#set_variable = { AAC_Active_Equipment_Array^AAC_Marine_Active_Category = equipment_id_temp }
	
	hidden_effect = {
		add_ideas = var:equipment_id_temp
	}

	set_temp_variable = { Equipment = equipment_id_temp }
	Antarctica_AAC_Marine_Assign_Equipment_Values = yes # look up the values for this equipment

	custom_effect_tooltip = Antarctica_AAC_Marine_Add_Equipment_Name_tt
	Antarctica_AAC_Marine_Equipment_Effect_Tooltip = yes #show a tooltip
	add_to_variable = { dirty_marines = 1 }

	if = {
		limit = {
			NOT = { is_in_array = { AAC_Unlocked_Equipment_Array = var:equipment_id_temp } }
		}
		add_to_array = { AAC_Unlocked_Equipment_Array = var:equipment_id_temp }
	}
}

Antarctica_AAC_Marine_Change_Equipment = {
	Antarctica_AAC_Marine_Remove_Current_Equipment = yes
	custom_effect_tooltip = AAC_equipment_will_be_added_tt
	Antarctica_AAC_Marine_Add_Equipment = yes

	# apply the change from values assigned in Antarctica_AAC_Marine_Add_Equipment
	add_to_variable = { AAC_Skill_Array^0 = mobility }
	add_to_variable = { AAC_Skill_Array^1 = endurance }
	add_to_variable = { AAC_Skill_Array^2 = aggression }
	add_to_variable = { AAC_Skill_Array^3 = loyalty }

	clamp_variable = { var = AAC_Skill_Array^0 min = 0 max = 100 }
	clamp_variable = { var = AAC_Skill_Array^1 min = 0 max = 100 }
	clamp_variable = { var = AAC_Skill_Array^2 min = 0 max = 100 }
	clamp_variable = { var = AAC_Skill_Array^3 min = 0 max = 100 }

	Antarctica_AAC_Marine_Skill_Check = yes

	if = {
		limit = { check_variable = { frame > 0 } }
		set_variable = { AAC_body_kit_frame = frame }
	}

	# Assign the equipment to be active
	set_variable = { AAC_Active_Equipment_Array^AAC_Marine_Active_Category = equipment_id_temp }
	add_to_variable = { dirty_marines = 1 }
}

	set_variable = { AAC_Diplomacy_Active_Country = state_id_temp }
	set_variable = { AAC_Diplomacy_Negotiation_Progress = neg_prog_temp }
	set_variable = { AAC_Diplomacy_Negotiation_Threshold = neg_threshold_temp }
	set_variable = { AAC_engagements_left = planned_engagements }
	subtract_from_variable = { AAC_engagements_left = 1 } # So the first engagement shows the remaining number correctly

	set_variable = { AAC_Diplomacy_Threshold_X = AAC_Diplomacy_Negotiation_Threshold }
	divide_variable = { AAC_Diplomacy_Threshold_X = 100 }
	multiply_variable = { AAC_Diplomacy_Threshold_X = 398 }

	set_country_flag = Antarctica_GUI_AAC_Diplomacy_Enabled

	add_to_array = { AAC_Diplo_Deal_Array = token:AAC_Diplo_Deal_Preamble }
	if = { 
		limit = {
			has_idea = AAC_Diplo_Deal_URG_OAS_Diplomatic_Offensive
		}
		add_to_variable = { AAC_Diplomacy_Negotiation_Progress = 2.5 }
	}
	if = {
		limit = {
			has_idea = AAC_Diplo_Deal_VEN_Independent_Focus
		}
		add_to_variable = { AAC_Diplomacy_Negotiation_Progress = -5 }
	}

	custom_effect_tooltip = AAC_diplo_setup_tt
}
