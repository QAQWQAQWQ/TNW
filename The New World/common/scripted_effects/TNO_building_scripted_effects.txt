#schools is weighted on pop heavily and GDP (small)
calculate_schools_bonus_state = {
	set_temp_variable = { total_state_factor = 1 }
	set_temp_variable = { b = 2 }
	set_temp_variable = { c = 3 } #total_state_factor+b
	multiply_temp_variable = { total_state_factor = local_gdp_percentage }
	multiply_temp_variable = { b = local_pop_percentage }
	add_to_temp_variable = { total_state_factor = b }
	divide_temp_variable = { total_state_factor = c }
	set_variable = { full_effect_schools_in_state = total_state_factor }

	if = { #KD's Notes - Every Place can use school coverage
		limit = {
			AND = {
				check_variable = { full_effect_schools_in_state < 0.005 }
				var:Prison_Scoping_State = { 
					check_variable = { building_level@schools = 0 } 
				}
			}
		}
		if = {
			limit = {
				OR = {
					var:Prison_Scoping_State = { has_state_category = small_island }
					var:Prison_Scoping_State = { has_state_category = tiny_island }
					var:Prison_Scoping_State = { has_state_category = enclave }
				}
			}
			set_variable = { full_effect_schools_in_state = 0.003 }
		}
		else = { set_variable = { full_effect_schools_in_state = 0.005 } }
	}
}

#offices is weighted on GDP heavily and on pop (small)
calculate_offices_bonus_state = {
	set_temp_variable = { total_state_factor = 3 }
	set_temp_variable = { b = 1 }
	set_temp_variable = { c = 4 } #total_state_factor+b
	multiply_temp_variable = { total_state_factor = local_gdp_percentage }
	multiply_temp_variable = { b = local_pop_percentage }
	add_to_temp_variable = { total_state_factor = b }
	divide_temp_variable = { total_state_factor = c }
	set_variable = { full_effect_offices_in_state = total_state_factor }

	#Amplify Based on Resource Deposits
	if = { #KD's Notes - You WOULD want these in resource-rich regions
		limit = {
			OR = {
				var:Prison_Scoping_State = { tungsten > 60 }
				var:Prison_Scoping_State = { oil > 60 }
				var:Prison_Scoping_State = { steel > 60 }
				var:Prison_Scoping_State = { chromium > 60 }
				var:Prison_Scoping_State = { rubber > 60 }
				var:Prison_Scoping_State = { aluminium > 60 }
				var:Prison_Scoping_State = { uranium > 60 }
			}
			AND = {
				check_variable = { full_effect_offices_in_state < 0.05 }
				var:Prison_Scoping_State = { 
					check_variable = { building_level@offices = 0 }
				}
			}
		}
		set_variable = { full_effect_offices_in_state = 0.05 }
	}
	else_if = { #Obviously you WOULD want these in resource-rich regions
		limit = {
			OR = {
				var:Prison_Scoping_State = { tungsten > 25 }
				var:Prison_Scoping_State = { oil > 25 }
				var:Prison_Scoping_State = { steel > 25 }
				var:Prison_Scoping_State = { chromium > 25 }
				var:Prison_Scoping_State = { rubber > 25 }
				var:Prison_Scoping_State = { aluminium > 25 }
				var:Prison_Scoping_State = { uranium > 25 }
				var:Prison_Scoping_State = {  has_dynamic_modifier = { modifier = KD_Large_Coal_Production } }
			}
			AND = {
				check_variable = { full_effect_offices_in_state < 0.03 }
				var:Prison_Scoping_State = { offices < 3 }
			}
		}
		set_variable = { full_effect_offices_in_state = 0.03 }
	}
	else_if = {
		limit = {
			OR = {
				var:Prison_Scoping_State = { tungsten > 5 }
				var:Prison_Scoping_State = { oil > 5 }
				var:Prison_Scoping_State = { steel > 5 }
				var:Prison_Scoping_State = { chromium > 5 }
				var:Prison_Scoping_State = { rubber > 5 }
				var:Prison_Scoping_State = { aluminium > 5 }
				var:Prison_Scoping_State = { uranium > 5 }
				var:Prison_Scoping_State = {  has_dynamic_modifier = { modifier = KD_Med_Coal_Production } }
			}
			AND = {
				check_variable = { full_effect_offices_in_state < 0.02 }
				var:Prison_Scoping_State = { offices < 2 }
			}
		}
		set_variable = { full_effect_offices_in_state = 0.02 }
	}
	else_if = {
		limit = {
			OR = {
				var:Prison_Scoping_State = { tungsten > 3 }
				var:Prison_Scoping_State = { oil > 3 }
				var:Prison_Scoping_State = { steel > 3 }
				var:Prison_Scoping_State = { chromium > 3 }
				var:Prison_Scoping_State = { rubber > 3 }
				var:Prison_Scoping_State = { aluminium > 3 }
				var:Prison_Scoping_State = { uranium > 3 }
				var:Prison_Scoping_State = {  has_dynamic_modifier = { modifier = KD_Minor_Coal_Production } }
			}
			AND = {
				check_variable = { full_effect_offices_in_state < 0.01 }
				var:Prison_Scoping_State = { 
					check_variable = { building_level@offices = 0 }
				}
			}
		}
		set_variable = { full_effect_offices_in_state = 0.01 }
	}
}

#Barracks are based on pop for coverage (would have liked on poverty too)
calculate_barracks_bonus_state = {
	set_variable = { full_effect_barracks_in_state = local_pop_percentage }
	if = {
		limit = {
			OR = {
				var:Prison_Scoping_State = { has_state_category = small_island }
				var:Prison_Scoping_State = { has_state_category = tiny_island }
			}
			NOT = { var:Prison_Scoping_State = { is_capital = yes } } #Prevents Edge-cases
		}
		multiply_variable = { full_effect_barracks_in_state = 5 }
		if = {
			limit = {
				AND = {
					check_variable = { full_effect_barracks_in_state < 0.01 }
					var:Prison_Scoping_State = { 
						check_variable = { building_level@barracks = 0 }
					}
				}
			}
			add_to_variable = { full_effect_barracks_in_state = 0.03 }
		}
		clamp_variable = {
			var = full_effect_barracks_in_state
			max = 5
		}
	}
	if = {
		limit = {
			OR = {
				var:Prison_Scoping_State = { has_state_category = enclave }
				var:Prison_Scoping_State = { has_state_flag = has_offshore_island } #States with this flag are likely much more developed
			}
			NOT = { var:Prison_Scoping_State = { is_capital = yes } } #Prevents Edge-cases
		}
		multiply_variable = { full_effect_barracks_in_state = 3 }
		if = {
			limit = {
				AND = {
					check_variable = { full_effect_barracks_in_state < 0.01 }
					var:Prison_Scoping_State = { 
						check_variable = { building_level@barracks = 0 } 
					}
				}
			}
			add_to_variable = { full_effect_barracks_in_state = 0.03 }
		}
		clamp_variable = {
			var = full_effect_barracks_in_state
			max = 5
		}
	}
}

debug_speed_test = {
	log = "start speed test"
	set_temp_variable = { var_counter = 0 }
	for_loop_effect = {
		end = 12
		value = v_0
		for_loop_effect = {
			end = 1000
			value = v_1

			for_loop_effect = {
				end = 1000
				value = v_2

				add_to_temp_variable = { var_counter = 2 }
				clamp_temp_variable = { var = var_counter min = 0 max = v_2 }
			}
		}
	}
	log = "break speed test var_counter [?var_counter]"
	set_temp_variable = { var_counter = 0 }
	for_loop_effect = {
		end = 12
		value = v_0
		for_loop_effect = {
			end = 1000
			value = v_1

			for_loop_effect = {
				end = 1000
				value = v_2

				add_to_temp_variable = { var_counter = 2 }
				if = {
					limit = { check_variable = { var_counter > v_2 } }
					set_temp_variable = { var_counter = v_2 }
				} else_if = {
					limit = { check_variable = { var_counter < 0 } }
					set_temp_variable = { var_counter = 0 }
				}
			}
		}
	}
	log = "end speed test var_counter [?var_counter]"
}
