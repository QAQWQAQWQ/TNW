TNO_ARG_FDP_Lira_Decline_Calc = {
	if = {
		limit = {
			check_variable = { clamped_inflation_rate > Previous_Inflation }
		}
		set_temp_variable = { current_temp_inflation = clamped_inflation_rate }
		divide_variable = { current_temp_inflation = Previous_Inflation  }
		multiply_variable = { TNO_ARG_FDP_Total_Decline = current_temp_inflation}
		set_country_flag = when_the_lira_is_collapsed
	}
	if = {
		limit = {
			check_variable = { clamped_inflation_rate < Previous_Inflation }
		}
		set_temp_variable = { current_temp_inflation = Previous_Inflation }
		divide_variable = { current_temp_inflation = clamped_inflation_rate  }
		multiply_variable = { TNO_ARG_FDP_Total_Decline = current_temp_inflation }
		set_country_flag = when_the_lira_is_grand
	}
	set_variable = { Previous_Inflation = clamped_inflation_rate }
}
