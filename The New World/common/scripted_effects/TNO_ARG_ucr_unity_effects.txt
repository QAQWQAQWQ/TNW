	clear_variable = ucr_total_percentage
	set_variable = {ucr_total_percentage = mid_influence}
	add_to_variable = {ucr_total_percentage = ud_influence}
	add_to_variable = {ucr_total_percentage = dp_influence}

ARG_calculate_others_mid = {
	set_temp_variable = { temp_variable_opposite_percentage = ud_influence }
	add_to_temp_variable = { temp_variable_opposite_percentage = dp_influence }

	set_variable = { ud_multiplier = ud_influence }
	divide_variable = { ud_multiplier = temp_variable_opposite_percentage }

	divide_variable = { dp_multiplier = temp_variable_opposite_percentage }

ARG_factional_var_check_mid = {
	ARG_total_percentage = yes
	set_temp_variable = { ucr_temp_variable_needs_adding = 100 }
	subtract_from_temp_variable = {ucr_temp_variable_needs_adding = ucr_total_percentage}
	set_variable = {ucr_needs_adding_multiplication = ucr_temp_variable_needs_adding}
	ARG_calculate_others_mid = yes
	if = {
		limit = {
			check_variable = {
				ucr_total_percentage > 100
			}
		}
		set_temp_variable = { ucr_temp_variable_needs_adding = ucr_needs_adding_multiplication }
		multiply_temp_variable = { ucr_temp_variable_needs_adding = ud_multiplier }
		add_to_variable = { ud_influence = ucr_temp_variable_needs_adding }

		set_temp_variable = { ucr_temp_variable_needs_adding = ucr_needs_adding_multiplication }
		multiply_temp_variable = { ucr_temp_variable_needs_adding = dp_multiplier }
		add_to_variable = { dp_influence = ucr_temp_variable_needs_adding }
	}
	else_if = {
		limit = {
			check_variable = {
				ucr_total_percentage < 100
			}
		}
		set_temp_variable = { ucr_temp_variable_needs_adding = ucr_needs_adding_multiplication }
		multiply_temp_variable = { ucr_temp_variable_needs_adding = dp_multiplier }
		add_to_variable = { dp_influence = ucr_temp_variable_needs_adding }

		set_temp_variable = { ucr_temp_variable_needs_adding = ucr_needs_adding_multiplication }
		multiply_temp_variable = { ucr_temp_variable_needs_adding = ud_multiplier }
		add_to_variable = { ud_influence = ucr_temp_variable_needs_adding }
	}
}

# Pie chart stuff, code stolen from Flax - Azhadron :wertroll:

UCR_GUI_piechart_refresh = {
	# Compute start and end values
	# MID is first so its start is 0 and end is just its %
	set_temp_variable = { mid_st = 0 }
	set_temp_variable = { mid_end = mid_influence }

	# Compute frame = SSSEEE by shifting over the start number 3 places
	# and adding the end number
	set_variable = { mid_frame = mid_st }
	round_variable = mid_frame
	multiply_variable = { mid_frame = 1000 }
	add_to_variable = { mid_frame = mid_end }

	# Compute start and end values

	set_temp_variable = { dp_st = mid_end }
	set_temp_variable = { dp_end = dp_st }
	add_to_temp_variable = { dp_end = dp_influence }

	set_variable = { dp_frame = dp_st }
	round_variable = dp_frame
	multiply_variable = { dp_frame = 1000 }
	add_to_variable = { dp_frame = dp_end }

	set_temp_variable = { ud_st = dp_end }
	set_temp_variable = { ud_end = ud_st }
	add_to_temp_variable = { ud_end = ud_influence }

	set_variable = { ud_frame = ud_st }
	round_variable = ud_frame
	multiply_variable = { ud_frame = 1000 }
	add_to_variable = { ud_frame = ud_end }
}
