MEX_setup_DW_connections = {
	# Setup connections to render
	clear_array = MEX_DW_rendered_connections_start
	clear_array = MEX_DW_rendered_connections_end

	for_each_loop = {
		array = MEX_DW_locations
		value = first_loc

		for_each_loop = {
			array = first_loc:MEX_DW_connections
			value = second_loc


			if = {
				limit = { check_variable = { first_loc < second_loc } }
				log = "[?first_loc] [?second_loc]"

				add_to_array = { MEX_DW_rendered_connections_start = first_loc }
				add_to_array = { MEX_DW_rendered_connections_end = second_loc }
			}
		}
	}

	clear_array = MEX_DW_rendered_connections_diff_x
	clear_array = MEX_DW_rendered_connections_diff_y
	clear_array = MEX_DW_rendered_connections_frame
	clear_array = MEX_DW_rendered_connections_f2s

	for_loop_effect = {
		start = 0
		end = MEX_DW_rendered_connections_start^num
		value = i

		set_temp_variable = { first_loc = MEX_DW_rendered_connections_start^i }
		set_temp_variable = { second_loc = MEX_DW_rendered_connections_end^i }

		if = {
			limit = { check_variable = { first_loc:MEX_DW_x > second_loc:MEX_DW_x } }
			set_temp_variable = { connection_x = second_loc:MEX_DW_x }
			set_temp_variable = { difference_x = first_loc:MEX_DW_x }
			subtract_from_temp_variable = { difference_x = second_loc:MEX_DW_x }

			if = {
				limit = { check_variable = { first_loc:MEX_DW_y > second_loc:MEX_DW_y } }
				set_temp_variable = { connection_y = second_loc:MEX_DW_y }
				set_temp_variable = { difference_y = first_loc:MEX_DW_y }
				subtract_from_temp_variable = { difference_y = second_loc:MEX_DW_y }
				set_temp_variable = { toggle = 0 }
				set_temp_variable = { f2s = 0 }
			}
			else = {
				set_temp_variable = { connection_y = first_loc:MEX_DW_y }
				set_temp_variable = { difference_y = second_loc:MEX_DW_y }
				subtract_from_temp_variable = { difference_y = first_loc:MEX_DW_y }
				set_temp_variable = { toggle = 1 }
				set_temp_variable = { f2s = 1 }
			}
		}
		else = {
			set_temp_variable = { connection_x = first_loc:MEX_DW_x }
			set_temp_variable = { difference_x = second_loc:MEX_DW_x }
			subtract_from_temp_variable = { difference_x = first_loc:MEX_DW_x }

			if = {
				limit = { check_variable = { first_loc:MEX_DW_y > second_loc:MEX_DW_y } }
				set_temp_variable = { connection_y = second_loc:MEX_DW_y }
				set_temp_variable = { difference_y = first_loc:MEX_DW_y }
				subtract_from_temp_variable = { difference_y = second_loc:MEX_DW_y }
				set_temp_variable = { toggle = 1 }
				set_temp_variable = { f2s = 0 }
			}
			else = {
				set_temp_variable = { connection_y = first_loc:MEX_DW_y }
				set_temp_variable = { difference_y = second_loc:MEX_DW_y }
				subtract_from_temp_variable = { difference_y = first_loc:MEX_DW_y }
				set_temp_variable = { toggle = 0 }
				set_temp_variable = { f2s = 1 }
			}
		}



		divide_temp_variable = { difference_x = 10 }
		divide_temp_variable = { difference_y = 10 }

		round_temp_variable = difference_x
		round_temp_variable = difference_y



		set_temp_variable = { connection_frame = toggle }
		multiply_temp_variable = { connection_frame = 1000 }
		round_temp_variable = connection_frame
		add_to_temp_variable = { connection_frame = difference_x }
		multiply_temp_variable = { connection_frame = 1000 }
		round_temp_variable = connection_frame
		add_to_temp_variable = { connection_frame = difference_y }

		add_to_array = { MEX_DW_rendered_connections_diff_x = connection_x }
		add_to_array = { MEX_DW_rendered_connections_diff_y = connection_y }
		add_to_array = { MEX_DW_rendered_connections_frame = connection_frame }
		add_to_array = { MEX_DW_rendered_connections_f2s = f2s }
	}
}

MEX_get_controlled_nodes = {
	#set_temp_variable = { loc_index = 0 }
	for_each_loop = {
		array = MEX_DW_locations
		value = loc

		clear_array = loc:MEX_DW_distance
		clear_array = loc:MEX_DW_supply_through

		set_temp_variable = { mex_dist_index = 0 }
		while_loop_effect = {
			limit = { check_variable = { mex_dist_index < MEX_DW_all_strongholds^num } }

			set_variable = { loc:MEX_DW_distance^mex_dist_index = -1 }
			set_variable = { loc:MEX_DW_supply_through^mex_dist_index = 0 }
			add_to_temp_variable = { mex_dist_index = 1 }
		}

		#set_variable = { loc:MEX_DW_distance^0 = -1 }

	}
	set_temp_variable = { new_stronghold_index = 0 }
	for_each_loop = {
		array = MEX_DW_all_strongholds
		value = stronghold

		set_temp_variable = { group = stronghold:MEX_DW_controller }

		clear_temp_array = to_explore
		clear_array = stronghold:MEX_DW_controlled_nodes
		clear_array = stronghold:MEX_DW_connected_nodes


		add_to_temp_array = { to_explore = stronghold }
		set_variable = { stronghold:MEX_DW_distance^new_stronghold_index = 0 }
		set_temp_variable = { MEX_DW_nodes_explored = 0 }
		add_to_array = { stronghold:MEX_DW_controlled_nodes = stronghold }

		while_loop_effect = {
			limit = {
				AND = {
					check_variable = { MEX_DW_nodes_explored < 7 }
					check_variable = { to_explore^num > 0 }
				}
			}

			set_temp_variable = { exploring = to_explore^0 }
			remove_from_temp_array = { array = to_explore index = 0 }


			for_each_loop = {
				array = exploring:MEX_DW_connections
				value = exploring_adjacent


				if = {
					limit = { check_variable = { exploring_adjacent:MEX_DW_controller = group } }

					if = {
						limit = { check_variable = { exploring_adjacent:MEX_DW_distance^new_stronghold_index = -1 } }

						set_variable = { exploring_adjacent:MEX_DW_parent^new_stronghold_index = exploring }

						set_variable = { exploring_adjacent:MEX_DW_distance^new_stronghold_index = exploring:MEX_DW_distance^new_stronghold_index }
						add_to_variable = { exploring_adjacent:MEX_DW_distance^new_stronghold_index = 1 }

						add_to_temp_array = { to_explore = exploring_adjacent }
						add_to_array = { stronghold:MEX_DW_controlled_nodes = exploring_adjacent }

					}
				}
				else = {

					if = {
						limit = { check_variable = { exploring_adjacent:MEX_DW_distance^new_stronghold_index = -1 } }
						set_variable = { exploring_adjacent:MEX_DW_distance^new_stronghold_index = exploring:MEX_DW_distance^new_stronghold_index }
						add_to_variable = { exploring_adjacent:MEX_DW_distance^new_stronghold_index = 1 }

						add_to_array = { stronghold:MEX_DW_connected_nodes = exploring_adjacent }

					}

				}
			}
			add_to_temp_variable = { MEX_DW_nodes_explored = 1 }
		}
	}
}

MEX_dw_node_traversal = {
	for_each_loop = {
		array = MEX_DW_locations
		value = loc

		set_variable = { loc:g_val = 1000 }
		set_variable = { loc:f_val = 0 }
	}

	clear_array = open_list
	clear_array = closed_list
	clear_array = path
	log = [?node_B]

	set_variable = { node_A:g_val = 0 }

	add_to_array = { open_list = node_A }

	while_loop_effect = {
		limit = {
			check_variable = { open_list^num > 0 }
		}
		break = break

		clear_variable = current_node
		set_variable = { min = open_list^0 }

		for_each_loop = {
			array = open_list
			value = opennodes

			if = {
				limit = {
					check_variable = { opennodes:f_val < min:f_val }
				}
				set_variable = { min = opennodes }
			}
		}

		set_variable = { current_node = min }

		remove_from_array = { array = open_list		value = min }

		if = {
			limit = {
				check_variable = { current_node = node_B }
			}
			log = [?current_node]
			while_loop_effect = {
				limit = {
					NOT = {
						check_variable = { current_node = node_A }
					}
				}
				add_to_array = { path = current_node }
				set_variable = { current_node = current_node:parent }
			}
			add_to_array = { path = current_node }
			set_temp_variable = { break = 1 }
		}

		add_to_array = { closed_list = current_node }

		for_each_loop = {
			array = current_node:MEX_DW_connections
			value = exploringnodes

			set_temp_variable = { node_1 = exploringnodes }
			set_temp_variable = { node_2 = node_A }
			MEX_DW_euclidean_dist = yes

			set_temp_variable = { tempg = eucdist }
			add_to_temp_variable = { tempg = current_node:g_val }

			if = {
				limit = {
					check_variable = { tempg < exploringnodes:g_val }
				}
				set_variable = { exploringnodes:parent = current_node }
				set_variable = { exploringnodes:g_val = tempg }
				add_to_variable = { exploringnodes:f_val = tempg }
				set_temp_variable = { node_1 = exploringnodes }
				set_temp_variable = { node_2 = node_B }
				MEX_DW_euclidean_dist = yes
				add_to_variable = { exploringnodes:f_val = eucdist }

				if = {
					limit = {
						NOT = {
							is_in_array = {
								array = closed_list
								value = exploringnodes
							}
						}
					}
					add_to_array = { open_list = exploringnodes }
				}
			}
		}
	}

MEX_DW_Should_Battle_Start = {

	if = {
		limit = {
			check_variable = { node_To:MEX_DW_node_units^num > 1 }
			check_variable = { node_To:MEX_DW_intel > 25 } #arb
			check_variable = { node_To:Is_In_Battle = 0 } #dont want a battle to keep restarting
		}

		set_temp_variable = { first_unit = node_To:MEX_DW_node_units^0 }
		set_temp_variable = { first_unit_alliegance = first_unit:MEX_DW_alliegance }

		for_each_loop = {
			array = node_To:MEX_DW_node_units
			value = checker_unit_val
			break = break

			if = {
				limit = {
					NOT = {
						check_variable = { checker_unit_val:MEX_DW_alliegance = first_unit_alliegance }
					}
				}
				set_variable = { node_To:MEX_DW_Battle_First_Group = first_unit_alliegance }
				set_variable = { node_To:MEX_DW_Battle_Second_Group = checker_unit_val:MEX_DW_alliegance }
				set_variable = { node_To:Is_In_Battle = 1 }
				set_temp_variable = { break = 1 }
			}
		}
		if = {
			limit = {
				check_variable = { node_To:Is_In_Battle = 1 }
			}
			MEX_DW_Organize_New_Battle = yes

			if = {
				limit = {
					check_variable = { node_To:Node_Not_Battle^num > 0 }
				}
				clear_array = node_To:Node_Not_Battle
				MEX_DW_update_gui_elements = yes
			}
		}
	}
}
MEX_DW_Organize_New_Battle = {
	add_to_array = { MEX_DW_Ongoing_Battles = node_To }

	for_each_loop = {
		array = node_To:MEX_DW_node_units
		value = unit_value

		log = "checking: [?unit_value]"

		if = {
			limit = {
				check_variable = { unit_value:MEX_DW_alliegance = node_To:MEX_DW_Battle_First_Group }
			}
			add_to_array = { node_To:MEX_DW_First_Group_Units = unit_value }
			if = {
				limit = {
					NOT = {
						is_in_array = {
							array = node_To:MEX_DW_First_Group_Unit_Types
							value = unit_value:MEX_DW_UNIT_TYPE
						}
					}
				}
				add_to_array = { node_To:MEX_DW_First_Group_Unit_Types = unit_value:MEX_DW_UNIT_TYPE }
			}
			
		}
		else_if = {
			limit = {
				check_variable = { unit_value:MEX_DW_alliegance = node_To:MEX_DW_Battle_Second_Group }
			}
			add_to_array = { node_To:MEX_DW_Second_Group_Units = unit_value }
			if = {
				limit = {
					NOT = {
						is_in_array = {
							array = node_To:MEX_DW_Second_Group_Unit_Types
							value = unit_value:MEX_DW_UNIT_TYPE
						}
					}
				}
				add_to_array = { node_To:MEX_DW_Second_Group_Unit_Types = unit_value:MEX_DW_UNIT_TYPE }
			}
		}
	}

	for_each_loop = { 
		array = node_To:MEX_DW_First_Group_Units
		value = u_valone

		add_to_temp_variable = { unitone_sum = u_valone:MEX_DW_node_num_units }
	}
	for_each_loop = { 
		array = node_To:MEX_DW_Second_Group_Units
		value = u_valone

		add_to_temp_variable = { unitwo_sum = u_valone:MEX_DW_node_num_units }
	}
	subtract_from_temp_variable = { unitone_sum = unitwo_sum }
	multiply_temp_variable = { unitone_sum = -0.01 }
	set_temp_variable = { input = 2.718 }
	set_temp_variable = { exp = unitone_sum }
	TNO_arith_exponential = yes
	add_to_temp_variable = { input = 1 }
	set_variable = { node_To:MEX_DW_Battle_leans = 50 }
	divide_variable = { node_To:MEX_DW_Battle_leans = input }
	add_to_variable = { node_To:MEX_DW_Battle_leans = 25 }
	#need to remember to recall MEX_DW_Should_Battle_Start at the end of the battle in case another faction walks in while your fighting another faction
}

	var:MEX_DW_selected_location = {
		clear_array = Node_Not_Battle

	}

	if = {
		limit = {
			check_variable = { MEX_DW_selected_location:Is_In_Battle = 0 }
		}
		var:MEX_DW_selected_location = {
			add_to_array = { Node_Not_Battle = 1 }
		}
	}
	else = {
		var:MEX_DW_selected_location = {
			add_to_array = { Node_In_Battle = 1 }
		}
	}
}

MEX_DW_euclidean_dist = {
	set_temp_variable = { xA = node_1:MEX_DW_x }
	set_temp_variable = { yA = node_1:MEX_DW_y }

	set_temp_variable = { xB = node_2:MEX_DW_x }
	set_temp_variable = { yB = node_2:MEX_DW_y }

	set_temp_variable = { xDiff = xB }
	subtract_from_temp_variable = { xDiff = xA }

	set_temp_variable = { yDiff = yB }
	subtract_from_temp_variable = { yDiff = yA }
	divide_temp_variable = { xDiff = 100 }
	divide_temp_variable = { yDiff = 100 }
	multiply_temp_variable = { xDiff = xDiff }

	multiply_temp_variable = { yDiff = yDiff }

	set_temp_variable = { dist_sqrd = 0 }
	add_to_temp_variable = { dist_sqrd = xDiff }
	add_to_temp_variable = { dist_sqrd = yDiff }

	set_temp_variable = { eucdist = dist_sqrd}
}
