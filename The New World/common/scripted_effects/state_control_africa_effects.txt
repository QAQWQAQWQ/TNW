	if = {
		limit = {
			OR = {
				controller = { is_in_array = { global.german_names = THIS } }
				AND = {
					is_controlled_by = COG
					Congo_is_rk = yes
				}
			}
			NOT = { controller = { has_country_leader = { character = GRO_Hans_Huttig ruling_only = yes } } }
		}
		set_state_name =  STATE_295_colonial
		set_province_name = { id = 5117 name = VICTORY_POINTS_5117_colonial }
		reset_province_name = 8193
		if = {
			limit = { has_global_flag = TNO_endonym_mode }
			set_state_name = STATE_295_ger
			set_province_name = { id = 5117 name = VICTORY_POINTS_5117_ger }
			set_province_name = { id = 8193 name =  VICTORY_POINTS_8193_ger }
		}
	}
	else_if = {
		limit = {
			controller = { is_in_array = { global.german_names = THIS } }
			controller = { has_country_leader = { character = GRO_Hans_Huttig ruling_only = yes } }
		}
		set_state_name = STATE_295_huttig
		set_province_name = { id = 5117 name = VICTORY_POINTS_5117_huttig }
		if = {
			limit = { has_global_flag = TNO_endonym_mode }
			set_province_name = { id = 8193 name =  VICTORY_POINTS_8193_ger }
		}
	}
	else_if = {
		limit = {
			OR = {
				is_controlled_by = SAF
				AND = {
					is_controlled_by = COG
					Congo_is_mandate = yes
				}
				AND = {
					is_controlled_by = USC
					CAR_is_mandate = yes
				}
				is_controlled_by = KTG
			}	
		}
		set_state_name =  STATE_295_colonial
		set_province_name = { id = 5117 name = VICTORY_POINTS_5117_colonial }
		reset_province_name = 8193
	}
	else = {
		reset_state_name = yes
		reset_province_name = 5117
		reset_province_name = 8193
		if = {
			limit = { has_global_flag = TNO_endonym_mode }
			set_province_name = { id = 8193 name =  VICTORY_POINTS_8193_cog }
		}
	}
}
