	TNO_admin_title_subdivisions_init = yes

	set_global_flag = TNO_admin_title_map_mode_admin_title_only
	
	every_country = {
		capital_scope = {
			if = {
				limit = {
					OR = {
						has_variable = TNO_admin_title
						has_variable = TNO_assoc_region 
					}
				}
				PREV = { set_variable = { TNO_selected_state_admin_title_map_mode = capital } }
			}
			else = {
				set_variable = { PREV.TNO_selected_state_admin_title_map_mode = 1 }
			}
		}
		force_update_map_mode = { mapmode = tno_admin_title_map_mode }
	}
}

## Handles any possible admin title interactions when a state switches owners
# Ex:
# if = {
#		limit = {
#			FROM = {
#				tag = <old owner>
#			}
#			ROOT = {
#				tag = <new owner>
#			}
#		}
#		FROM.FROM = {
#			<effects here>
#		}
#	}
TNO_admin_title_state_control_effect = { 
	log = "TNO_admin_title_state_control_effect - State: [FROM.FROM.GetID] - ROOT: [old_owner.GetTag] -> FROM: [new_owner.GetTag]"
	if = {
		limit = {
			OR = {
				FROM = {
					tag = BKU
				}
				AND = {
					FROM = {
						tag = TUR
					}
					OR = {
						has_global_flag = ita_tur_ita_win
						has_global_flag = ita_tur_ita_lose
					}
				}
			}
			ROOT = {
				OR = {
					is_in_faction_with = IME
					tag = IRQ
				}
			}
		}
		FROM.FROM = {
			clear_variable = TNO_admin_title
			clear_variable = TNO_assoc_region
			if = {
				limit = {
					ROOT = {
						tag = IRQ
					}
					FROM.FROM = {
						OR = {
							is_claimed_by = IRQ
							is_core_of = IRQ
						}
					}
				}
				FROM.FROM = { 
					set_temp_variable = { admin_title_temp = token:Admin_Title_Banner }
					TNO_adjust_admin_title = yes 
				}
			}
		}
	}
	else_if = {
		limit = {
			ROOT = {
				OR = {
					is_in_faction_with = TUR
					tag = BKU
				}
			}
		}
		if = {
			limit = {
				FROM.FROM = {
					is_core_of = BKU
				}
			}
			FROM.FROM = { 
				set_temp_variable = { assoc_region_temp = token:Assoc_Region_Southeast_Anatolia }
				TNO_adjust_assoc_region = yes 
			}
			if = {
				limit = {
					FROM.FROM = {
						check_variable = { TNO_assoc_region = token:Assoc_Region_Southeast_Anatolia }
						NOT = {
							state = 1779
							state = 1993
						}
					}
