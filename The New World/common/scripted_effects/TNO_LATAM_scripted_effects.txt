		limit = { check_variable = { GUC.anti_brazil_sentiment_temp < 0 } }
		multiply_temp_variable = { amount = -1 }
		custom_effect_tooltip = BRA_cayenne_public_anger_decrease_tt
	}
	else = {
		custom_effect_tooltip = BRA_cayenne_public_anger_increase_tt

	clamp_variable = {
		var = GUC.anti_brazil_sentiment
		min = 0
		max = 100
	}

BRA_cayenne_calculate_ending = {
	if = { #Disaster Scenario
		limit = {
			OR = {
				AND = {
					check_variable = {GUC.anti_brazil_sentiment > 60}
					check_variable = {GUC.indep_turnout > GUC.brazilian_turnout}
					has_game_rule = {
						rule = GUC_INDEPENDENCE
						option = DEFAULT
					}
				}
				has_game_rule = {
					rule = GUC_INDEPENDENCE
					option = GUC_INDEPENDENCE_DISASTER
				}
			}
		}
		country_event = {
			id = LATAM_cayenne_endings.20
			days = 1
		}
	}
	else_if = { #Lose Referendum Scenario
		limit = {
			OR = {
				AND = {
					check_variable = {GUC.indep_turnout > GUC.brazilian_turnout}
					has_game_rule = {
						rule = GUC_INDEPENDENCE
						option = DEFAULT
					}
				}
				has_game_rule = {
					rule = GUC_INDEPENDENCE
					option = GUC_INDEPENDENCE_LOSE
				}
			}
		}
		country_event = {
			id = LATAM_cayenne_endings.17
			days = 1
		}
	}
	else_if = { #Public Anger Fail Scenario
		limit = {
			OR = {
				AND = {
					check_variable = {GUC.anti_brazil_sentiment > 60}
					check_variable = {GUC.indep_turnout < GUC.brazilian_turnout}
					has_game_rule = {
						rule = GUC_INDEPENDENCE
						option = DEFAULT
					}
				}
				has_game_rule = {
					rule = GUC_INDEPENDENCE
					option = GUC_INDEPENDENCE_ANGER
				}
			}
		}
		country_event = {
			id = LATAM_cayenne_endings.8
			days = 1
		}
	}
	else_if = { #Legitimacy Fail Scenario
		limit = {
			OR = {
				AND = {
					check_variable = {GUC.referendum_legitimacy < 40}
					check_variable = {GUC.indep_turnout < GUC.brazilian_turnout}
					has_game_rule = {
						rule = GUC_INDEPENDENCE
						option = DEFAULT
					}
				}
				has_game_rule = {
					rule = GUC_INDEPENDENCE
					option = GUC_INDEPENDENCE_LEGITIMACY
				}
			}
		}
		country_event = {
			id = LATAM_cayenne_endings.11
			days = 1
		}
	}
	else_if = { #Golden Ending Scenario
		limit = {
			OR = {
				AND = {
					check_variable = {GUC.anti_brazil_sentiment < 30}
					check_variable = {GUC.referendum_legitimacy > 70}
					check_variable = {GUC.indep_turnout < GUC.brazilian_turnout}
					has_game_rule = {
						rule = GUC_INDEPENDENCE
						option = DEFAULT
					}
				}
				has_game_rule = {
					rule = GUC_INDEPENDENCE
					option = GUC_INDEPENDENCE_GOLDEN
				}
			}
		}
		country_event = {
			id = LATAM_cayenne_endings.1
			days = 1
		}
	}
	else_if = { #Good Ending Scenario
		limit = {
			OR = {
				AND = {
					check_variable = {GUC.indep_turnout < GUC.brazilian_turnout}
					has_game_rule = {
						rule = GUC_INDEPENDENCE
						option = DEFAULT
					}
				}
				has_game_rule = {
					rule = GUC_INDEPENDENCE
					option = GUC_INDEPENDENCE_GOOD
				}
			}
		}
		country_event = {
			id = LATAM_cayenne_endings.4
			days = 1
		}
	}
	BRA = {
		set_temp_variable = { alert_id = token:Alert_LATAM_Cayenne_Referendum }
		remove_TNO_alert = yes
	}
	USA = {
		clr_country_flag = USA_CIA_cayenne_adhemar
	}
}
