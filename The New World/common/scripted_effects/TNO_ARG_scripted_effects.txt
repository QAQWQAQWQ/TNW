ARG_Onga_Check = {
	custom_effect_tooltip = ARG_The_Scales_Will_Tip

	if = {
		limit = { has_country_flag = ARG_Liberal_Onga }
		country_event = {
			id = ARG.119
			days = 1
		}
	}
	else_if = {
		limit = { has_country_flag = ARG_Peron_Returns }
		country_event = {
			id = ARG.11
			days = 3
		}
	}
	else_if = {
		limit = { has_country_flag = ARG_Middle_Road }
		country_event = {
			id = ARG.121
			days = 1
		}
	}
}
