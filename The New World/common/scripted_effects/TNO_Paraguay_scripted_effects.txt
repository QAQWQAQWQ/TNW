PAR_check_states = {
	for_each_loop = {
		array = paraguay_map_grid_array
		if = {
			limit = {
				var:v = {
					check_variable = { PAR_Government_Control > PAR_Communist_Control }
				}
			}
			var:v = { set_variable = { state_condition = 1 } }
		}
		else_if = {
			limit = {
				var:v = {
					check_variable = { PAR_Communist_Control > PAR_Government_Control }
				}
			}
			var:v = { set_variable = { state_condition = 2 } }
		}
		else_if = {
			limit = {
				var:v = {
					check_variable = { PAR_Communist_Control = PAR_Government_Control }
				}
			}
			var:v = { set_variable = { state_condition = 3 } }
		}
	}
}
PAR_bush_war_calculate_stability = {
	set_variable = { PAR.PAR_regime_stability = 100 }
	clamp_variable = {
		var = PAR.regime_minus
		min = -100
		max = 0
	}
	add_to_variable = { PAR.PAR_regime_stability = PAR.regime_minus }

	log = "regime_minus: [?PAR.regime_minus]"
}
TNO_PAR_Calculate_Frame = {
	for_each_loop = {
		array = paraguay_map_grid_array
		var:v = {
			if = {
				limit = {
					check_variable = { PAR_Government_Control < PAR_Communist_Control }
				}
				if = {
					limit = {
						check_variable = { state_condition = 1 }
					}
					add_to_variable = { PAR.regime_minus = -5 }
					PAR_bush_war_calculate_stability = yes
				}
				set_variable = { state_condition = 2 }

				set_variable = { gui_display_frame = state_condition }
				multiply_variable = { gui_display_frame = 1000 }

				set_temp_variable = { gui_display_frame_add = PAR_Communist_Control }
				subtract_from_temp_variable = { gui_display_frame_add = PAR_Government_Control }
				divide_temp_variable = { gui_display_frame_add = PAR_Communist_Control }
				multiply_temp_variable = { gui_display_frame_add = 100 }

				add_to_variable = { gui_display_frame = gui_display_frame_add }

				set_demilitarized_zone = yes
			}
			else = {
				if = {
					limit = {
						check_variable = { state_condition = 2 }
					}
					add_to_variable = { PAR.regime_minus = 5 }
					PAR_bush_war_calculate_stability = yes
				}
				set_variable = { state_condition = 1 }

				set_variable = { gui_display_frame = state_condition }
				multiply_variable = { gui_display_frame = 1000 }

				set_temp_variable = { gui_display_frame_add = PAR_Government_Control }
				subtract_from_temp_variable = { gui_display_frame_add = PAR_Communist_Control }
				divide_temp_variable = { gui_display_frame_add = PAR_Government_Control }
				multiply_temp_variable = { gui_display_frame_add = 100 }

				add_to_variable = { gui_display_frame = gui_display_frame_add }

				set_demilitarized_zone = no
			}
		}
	}
	log = "[?PAR_selected_state:gui_display_frame]"
}

		limit = { has_global_flag = PAR_bush_war_government_action_flag }

	}
	else = {
		set_global_flag = PAR_bush_war_government_action_flag

		limit = { check_variable = { PAR_government_action_var = 0 } }
		set_temp_variable = { PAR_control_temp = 10 }

				}
				25 = {
					set_variable = { PAR_government_action_var = 1 } # Increase Government Support By Twenty
					set_variable = { PAR_government_action_region_var = 1616 }
				}
			}
		}
		else = {
			random_list = {
				60 = {
					set_variable = { PAR_government_action_var = 0 } # Increase Government Support By Ten
					TNO_PAR_bush_war_random_region_government_decide = yes

				20 = {
					set_variable = { PAR_government_action_var = 1 } # Increase Government Support By Twenty

TNO_PAR_bush_war_fulna_take_over_check = {
	set_variable = { PAR.par_fulna = 0 }

	for_each_loop = {
		array = paraguay_map_grid_array
		if = {
			limit = { var:v = { check_variable = { state_condition = 2 } } }
			add_to_variable = { PAR.par_fulna = 1 }
		}
	}

	if = { # FULNA Takes Over
		limit = {
			OR = {
				check_variable = { PAR.par_fulna > 3.9 }
				1616 = { check_variable = { state_condition = 2 } }
			}
		}
		if = {
			limit = { NOT = { has_global_flag = PAR_bush_war_fulna_take_over_imminent } }
			set_global_flag = PAR_bush_war_fulna_take_over_imminent
			ZZZ = { activate_mission = ZZZ_BRA_bush_war_fulna_take_over_imminent }
		}
	}

	else = {
		clr_global_flag = PAR_bush_war_fulna_take_over_imminent
	}
}
