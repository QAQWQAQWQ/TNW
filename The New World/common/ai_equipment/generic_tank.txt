#The first thing to know is the AI simply won't use a chassis without a defined "template", thus this file is very important

### IFV
generic_IFV = { #First we name the set of "templates" we are about to design
	category = land #Then we define whether it is for tanks (land) or ships (naval), TNO does not use the air designer

	available_for = {} #This block defines what countries can use it, you enter their tag here. If kept empty, every tag can use it

	roles = { #This is the role of the template, for the AI to use it, it will need to match the role of the division templates you want the equipment to be used in
		armor
		land_light_tank
	}

	priority = { #This block determines how important this particular set of templates is compared to others
		factor = 100
		modifier = {
			factor = 15
			has_template_containing_unit = IFV
		}
		modifier = {
			factor = 0
			NOT = { has_template_containing_unit = IFV }
		}
	}

	early_IFV_default = { #Every individual template needs a name as well, for simplicity, use the same ones as the equipment
		priority = { #This will determine the priority of this particular template compared to others in the set
			factor = 35
			modifier = { #You can use modifiers to change the priority of the template
				OR = {
					has_tech = NSB_IFV_1950
					has_tech = NSB_MBT_1950
				}
				factor = 0
			}
		}

		target_variant = { 		#Here we define how the actual template will look
			match_value = 1000 		#This is the value of the template when finished, the AI will construct the ones with the highest value
			type = IFV_chassis_0 	#This block defines the particular equipment/chassis to be used
			modules = { 			#From he we define the individual modules the template will use
				main_armament_slot = { 		#All tanks require a main armament
					any_of = { 				#I recommend using any_of and within a list of the specific equipment you want in the slot, slotted from best to worst
						IFV_cannon_4
						IFV_cannon_3
						IFV_cannon_2
						IFV_cannon
					}
					upgrade > current 		#This block makes it so the AI will automatically upgrade to better versisons of the module category if available
				}
				turret_type_slot = IFV_three_man_tank_turret
				suspension_type_slot = tank_torsion_bar_suspension
				armor_type_slot = tank_welded_armor

						smoke_emitter

				special_type_slot_4 = {
					any_of = {
						fire_control_system
						stabilizer
						additional_machine_guns
					}
					upgrade > current
				}
			}
			upgrades = { 	#This block defines the value of the upgrades that can be added on to the build
				tank_nsb_engine_upgrade = 1 #Remember to make sure the template has enough speed to be valid. It needs, at the minimum, 5km per hour.
				tank_nsb_armor_upgrade = 2
			}
		}
		allowed_modules = { #In this block, you list all modules the AI can use to fill the empty slots. Any modules not on this list will not be used. The AI will always prioritize the modules at the top of the list.
			IFV_cannon_4
			IFV_cannon_3
			IFV_cannon_2
			IFV_cannon
